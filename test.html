$("#total-cost-button").click(function() { var selectedColor = $("#color-filter").val(); var totalCost = 0; // Calculate total cost based on color filter if (selectedColor === "all") { // No color filter, calculate cost for all rows $(".condition-1, .condition-2").each(function() { totalCost += parseFloat($(this).find("td:nth-child(6)").text().replace(/\$/, '')); }); } else { // Color filter applied, calculate cost for matching rows $("." + selectedColor).each(function() { totalCost += parseFloat($(this).find("td:nth-child(6)").text().replace(/\$/, '')); }); } // Display the total cost $("#total-cost").text(totalCost.toFixed(2)); });
